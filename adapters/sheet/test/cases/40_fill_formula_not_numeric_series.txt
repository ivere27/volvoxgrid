# Formula-filled ranges should shift formulas, not convert to numeric series
setData [["1","=A1"],["2","=A2"],["3",""],["4",""]]
setSelection {"row1":0,"col1":1,"row2":1,"col2":1}
invokeFillTo 3 1
assertCellRawValue 2 1 "=A3"
assertCellRawValue 3 1 "=A4"
assertCellValue 2 1 "3"
assertCellValue 3 1 "4"
